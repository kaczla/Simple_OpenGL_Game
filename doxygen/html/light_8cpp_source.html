121 void Light::BindVAO(){
122  if( this->Init ){
123  SDL_Log( "Binding %s into VAO\n", this->OBJPathFile.c_str() );
124  glGenVertexArrays( 1, &this->VAO );
125  glGenBuffers( 1, &this->VertexBuffer );
126  glGenBuffers( 1, &this->IndicesBuffer );
127 
128  //Vertex:
129  glBindBuffer( GL_ARRAY_BUFFER, this->VertexBuffer );
130  glBufferData( GL_ARRAY_BUFFER, this->Vertices.size() * sizeof( glm::vec3 ), &this->Vertices[0], GL_STATIC_DRAW );
131 
132  //VAO:
133  glBindVertexArray( this->VAO );
134 
135  //Vertex:
136  glBindBuffer( GL_ARRAY_BUFFER, this->VertexBuffer );
137  glVertexAttribPointer( 0, 3, GL_FLOAT, GL_FALSE, 0, (GLvoid *)0 );
138  glEnableVertexAttribArray( 0 );
139  //Indicies:
140  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, this->IndicesBuffer );
141  glBufferData( GL_ELEMENT_ARRAY_BUFFER, this->Indices.size() * sizeof(GLuint), &this->Indices[0], GL_STATIC_DRAW );
142 
143  glBindVertexArray( 0 );
144 
145  SDL_Log( "Binded %s into VAO\n", this->OBJPathFile.c_str() );
146  }
147  else{
148  SDL_LogError( SDL_LOG_CATEGORY_ERROR, "Before binding, load file and texture!\n" );
149  }
150 }
