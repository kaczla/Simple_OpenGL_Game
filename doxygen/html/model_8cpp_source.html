253 void Model::BindVAO(){
254  if( this->Init ){
255  SDL_Log( "Binding %s into VAO\n", this->OBJPathFile.c_str() );
256  glGenVertexArrays( 1, &this->VAO );
257  glGenBuffers( 1, &this->VertexBuffer );

260  glGenBuffers( 1, &this->IndicesBuffer );
261 
262  //Vertex:
263  glBindBuffer( GL_ARRAY_BUFFER, this->VertexBuffer );
264  glBufferData( GL_ARRAY_BUFFER, this->Vertices.size() * sizeof( glm::vec3 ), &this->Vertices[0], GL_STATIC_DRAW );
265  //Uv:
266  glBindBuffer( GL_ARRAY_BUFFER, this->UvBuffer );
267  glBufferData( GL_ARRAY_BUFFER, this->Uvs.size() * sizeof( glm::vec2 ), &this->Uvs[0], GL_STATIC_DRAW );
268  //Normal:
269  glBindBuffer( GL_ARRAY_BUFFER, this->NormalBuffer );
270  glBufferData( GL_ARRAY_BUFFER, this->Normals.size() * sizeof( glm::vec3 ), &this->Normals[0], GL_STATIC_DRAW );
271 
272  //VAO:
273  glBindVertexArray( this->VAO );
274 
275  //Vertex:
276  glBindBuffer( GL_ARRAY_BUFFER, this->VertexBuffer );
277  glVertexAttribPointer( 0, 3, GL_FLOAT, GL_FALSE, 0, (GLvoid *)0 );
278  glEnableVertexAttribArray( 0 );
279  //Uv:
280  glBindBuffer( GL_ARRAY_BUFFER, this->UvBuffer );
281  glVertexAttribPointer( 1, 2, GL_FLOAT, GL_FALSE, 0, (GLvoid *)0 );
282  glEnableVertexAttribArray( 1 );
283  //Normal:
284  glBindBuffer( GL_ARRAY_BUFFER, this->NormalBuffer );
285  glVertexAttribPointer( 2, 3, GL_FLOAT, GL_FALSE, 0, (GLvoid *)0 );
286  glEnableVertexAttribArray( 2 );
287  //Indicies:
288  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, this->IndicesBuffer );
289  glBufferData( GL_ELEMENT_ARRAY_BUFFER, this->Indices.size() * sizeof(GLuint), &this->Indices[0], GL_STATIC_DRAW );
290 
291  glBindVertexArray( 0 );
