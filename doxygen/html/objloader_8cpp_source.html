221 bool LoadAssimp( const char *path_file,
222  std::vector <glm::vec3> &vertices,

224  std::vector <glm::vec3> &normals,

228  uvs.clear();
229  normals.clear();
230  indices.clear();
231  SDL_Log( "Loading file: %s\n", path_file );
232  Assimp::Importer importer;
233  const aiScene* scene = importer.ReadFile( path_file, 0 );
234  if( !scene ){
235  SDL_LogError( SDL_LOG_CATEGORY_INPUT, "importer.ReadFile: %s\n", importer.GetErrorString() );
236  return false;
237  }
238  const aiMesh* mesh = scene->mMeshes[0];
239  unsigned int i = 0;
240  aiVector3D tmp;
241  //vertices:
242  vertices.reserve( mesh->mNumVertices );
243  for( i = 0; i < mesh->mNumVertices; ++i ){
244  tmp = mesh->mVertices[i];
245  vertices.push_back( glm::vec3( tmp.x, tmp.y, tmp.z ) );
246  }
247  //uvs:
248  uvs.reserve( mesh->mNumVertices );
249  for( i = 0; i < mesh->mNumVertices; ++i ){
250  tmp = mesh->mTextureCoords[0][i];
251  //important!
252  //This is done because most images have the top y-axis inversed with OpenGL's top y-axis.
253  //or conver in shader
254  uvs.push_back( glm::vec2( tmp.x, 1.0 - tmp.y ) );
255  }
256  //normals:
257  normals.reserve( mesh->mNumVertices );
258  for( i = 0; i < mesh->mNumVertices; ++i ){
259  tmp = mesh->mNormals[i];
260  normals.push_back( glm::vec3( tmp.x, tmp.y, tmp.z ) );
261  }
262  //indices:
263  indices.reserve( 3*mesh->mNumFaces );
264  for( i = 0; i < mesh->mNumFaces; ++i ){
265  indices.push_back( mesh->mFaces[i].mIndices[0] );
266  indices.push_back( mesh->mFaces[i].mIndices[1] );
267  indices.push_back( mesh->mFaces[i].mIndices[2] );
268  }
269  SDL_Log( "vertex:%u uv:%u normal:%u indices:%u\n", vertices.size(), uvs.size(), normals.size(), indices.size() );
270  SDL_Log( "Loaded file: %s\n", path_file );
271  return true;
272 }
